C++ / Arduino — ESP32 Full Firmware 
// ──────────────────────────────────────────────────────
// Smart HVAC — ESP32 Full Firmware
// Project: Hybrid Control Loop Architecture
// Hardware: ESP32 + DHT22 + 4-CH Relay + OLED SSD1306
// ──────────────────────────────────────────────────────

#include <WiFi.h>
#include <PubSubClient.h>
#include <DHT.h>
#include <ArduinoJson.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

// ── Pin Definitions ───────────────────────────────────
#define DHTPIN       4
#define DHTTYPE     DHT22
#define RELAY_AC    16   // Air conditioner
#define RELAY_FAN   17   // Ventilation fan
#define RELAY_DH    18   // Dehumidifier
#define RELAY_HEAT  19   // Heater
#define OLED_SDA    21
#define OLED_SCL    22

// ── Network & MQTT Config ──────────────────────────────
const char* WIFI_SSID   = "YourWiFiName";
const char* WIFI_PASS   = "YourWiFiPassword";
const char* MQTT_SERVER = "192.168.1.100";
const int   MQTT_PORT   = 1883;
const char* CLIENT_ID   = "esp32-hvac-01";

// Topics
const char* TOPIC_PUB    = "hvac/sensors";
const char* TOPIC_CMD    = "hvac/cmd";
const char* TOPIC_STATUS = "hvac/status";

// ── Control Parameters ─────────────────────────────────
float  setpoint    = 22.0;  // Target °C
float  hysteresis = 1.0;   // ±°C dead-band
float  humTarget  = 55.0;  // Target %RH
int    fanSpeed   = 60;    // Fan % (0-100)
bool   cloudMode  = true;  // Enable cloud optimization

// ── Instances ──────────────────────────────────────────
DHT               dht(DHTPIN, DHTTYPE);
WiFiClient        espClient;
PubSubClient      mqttClient(espClient);
Adafruit_SSD1306  display(128, 64, &Wire);

unsigned long lastPublish  = 0;
unsigned long lastSample   = 0;
float         lastTemp     = 0;
float         lastHum      = 0;

// ── WiFi Setup ─────────────────────────────────────────
void setupWiFi() {
  WiFi.begin(WIFI_SSID, WIFI_PASS);
  Serial.print("Connecting to WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi OK — IP: " + WiFi.localIP().toString());
}

// ── MQTT Callback (receive cloud commands) ─────────────
void mqttCallback(char* topic, byte* payload, unsigned int length) {
  StaticJsonDocument<256> doc;
  if (deserializeJson(doc, payload, length) != DeserializationError::Ok) return;

  if (doc.containsKey("setpoint"))  setpoint   = doc["setpoint"].as<float>();
  if (doc.containsKey("fan"))       fanSpeed   = doc["fan"].as<int>();
  if (doc.containsKey("hum_target")) humTarget  = doc["hum_target"].as<float>();

  // Direct relay control from cloud
  if (doc.containsKey("ac"))   digitalWrite(RELAY_AC,   doc["ac"]   == "on" ? HIGH : LOW);
  if (doc.containsKey("fan_on")) digitalWrite(RELAY_FAN, doc["fan_on"] ? HIGH : LOW);

  Serial.printf("Cloud CMD: setpoint=%.1f fan=%d%%\n", setpoint, fanSpeed);
}

// ── MQTT Reconnect ─────────────────────────────────────
void mqttReconnect() {
  while (!mqttClient.connected()) {
    Serial.print("MQTT connecting...");
    if (mqttClient.connect(CLIENT_ID)) {
      mqttClient.subscribe(TOPIC_CMD);
      Serial.println("OK");
    } else {
      Serial.printf("Failed rc=%d — retry in 5s\n", mqttClient.state());
      delay(5000);
    }
  }
}

// ── LOCAL CONTROL LOOP ─────────────────────────────────
// This runs every loop() — no cloud required
// Priority: local always overrides cloud in emergencies
void localControlLoop(float temp, float hum) {
  // Temperature control with hysteresis dead-band
  if (temp > setpoint + hysteresis) {
    digitalWrite(RELAY_AC,   HIGH);  // AC ON
    digitalWrite(RELAY_FAN,  HIGH);  // Fan ON
    digitalWrite(RELAY_HEAT, LOW);   // Heater OFF
    Serial.println("[LOCAL] COOLING — AC ON");
  }
  else if (temp < setpoint - hysteresis) {
    digitalWrite(RELAY_AC,   LOW);   // AC OFF
    digitalWrite(RELAY_HEAT, HIGH);  // Heater ON
    digitalWrite(RELAY_FAN,  HIGH);  // Circulate
    Serial.println("[LOCAL] HEATING — Heater ON");
  }
  else {
    digitalWrite(RELAY_AC,   LOW);
    digitalWrite(RELAY_HEAT, LOW);
    Serial.println("[LOCAL] IDLE — within setpoint");
  }

  // Humidity control
  if (hum > humTarget + 5.0) {
    digitalWrite(RELAY_DH, HIGH);
    Serial.println("[LOCAL] Dehumidifier ON");
  } else if (hum < humTarget - 5.0) {
    digitalWrite(RELAY_DH, LOW);
  }
}

// ── CLOUD LOOP — Publish sensor data ──────────────────
void publishSensorData(float temp, float hum) {
  StaticJsonDocument<256> doc;
  doc["device_id"] = CLIENT_ID;
  doc["temp"]      = round(temp * 10.0) / 10.0;
  doc["hum"]       = round(hum);
  doc["ac_on"]     = digitalRead(RELAY_AC);
  doc["fan_on"]    = digitalRead(RELAY_FAN);
  doc["setpoint"]  = setpoint;
  doc["uptime"]    = millis() / 1000;

  char buf[256];
  serializeJson(doc, buf);
  mqttClient.publish(TOPIC_PUB, buf);
  Serial.printf("[CLOUD] Published: T=%.1f H=%.0f\n", temp, hum);
}

// ── OLED Display Update ────────────────────────────────
void updateDisplay(float temp, float hum) {
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);

  // Title
  display.setTextSize(1);
  display.setCursor(0, 0);
  display.print("Smart HVAC v2.1");
  display.drawLine(0, 9, 128, 9, SSD1306_WHITE);

  // Temperature — large
  display.setTextSize(2);
  display.setCursor(0, 14);
  display.printf("%.1fC", temp);

  // Humidity
  display.setTextSize(1);
  display.setCursor(0, 36);
  display.printf("Hum: %.0f%%", hum);

  // Status
  display.setCursor(0, 48);
  display.print("Set:");
  display.print(setpoint);
  display.print("  AC:");
  display.print(digitalRead(RELAY_AC) ? "ON" : "off");

  // WiFi indicator
  display.setCursor(100, 0);
  display.print(WiFi.status() == WL_CONNECTED ? "WiFi" : "----");

  display.display();
}

// ── SETUP ──────────────────────────────────────────────
void setup() {
  Serial.begin(115200);

  // Relay pins — active HIGH
  pinMode(RELAY_AC,   OUTPUT); digitalWrite(RELAY_AC,   LOW);
  pinMode(RELAY_FAN,  OUTPUT); digitalWrite(RELAY_FAN,  LOW);
  pinMode(RELAY_DH,   OUTPUT); digitalWrite(RELAY_DH,   LOW);
  pinMode(RELAY_HEAT, OUTPUT); digitalWrite(RELAY_HEAT, LOW);

  dht.begin();

  // OLED init
  Wire.begin(OLED_SDA, OLED_SCL);
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("OLED init failed");
  }
  display.clearDisplay();
  display.display();

  setupWiFi();
  mqttClient.setServer(MQTT_SERVER, MQTT_PORT);
  mqttClient.setCallback(mqttCallback);
  mqttClient.setBufferSize(512);

  Serial.println("Setup complete — entering main loop");
}

// ── MAIN LOOP ──────────────────────────────────────────
void loop() {
  // Maintain MQTT connection
  if (!mqttClient.connected()) mqttReconnect();
  mqttClient.loop();

  unsigned long now = millis();

  // Sample sensors every 2 seconds
  if (now - lastSample >= 2000) {
    lastSample = now;
    float temp = dht.readTemperature();
    float hum  = dht.readHumidity();

    if (!isnan(temp) && !isnan(hum)) {
      lastTemp = temp;
      lastHum  = hum;

      // LOCAL CONTROL LOOP — runs every 2s regardless of WiFi
      localControlLoop(temp, hum);
      updateDisplay(temp, hum);
    } else {
      Serial.println("DHT read failed — check wiring");
    }
  }

  // Publish to cloud every 10 seconds (CLOUD LOOP)
  if (now - lastPublish >= 10000 && mqttClient.connected()) {
    lastPublish = now;
    publishSensorData(lastTemp, lastHum);
  }
}

# ──────────────────────────────────────────────────────
# Smart HVAC Cloud Server
# Install: pip install paho-mqtt flask flask-cors
# Run:     python cloud_server.py
# ──────────────────────────────────────────────────────

import paho.mqtt.client as mqtt
import json, time, statistics, threading
from flask import Flask, jsonify, request
from flask_cors import CORS
from datetime import datetime
from collections import deque

# ── Config ─────────────────────────────────────────────
BROKER_HOST  = "localhost"
BROKER_PORT  = 1883
TOPIC_SUB    = "hvac/sensors"
TOPIC_CMD    = "hvac/cmd"
WINDOW_SIZE  = 10    # readings before optimizing
PUBLISH_EVERY = 5   # optimize every N readings

# ── Global state ───────────────────────────────────────
readings    = deque(maxlen=100)   # ring buffer of sensor data
publish_cnt = 0
mqtt_client = None

# ── Global optimizer ───────────────────────────────────
def optimize_setpoint(recent):
  """
  Simple rule-based optimizer. Replace with ML model
  (scikit-learn, TensorFlow Lite) for production.
  """
  temps  = [r["temp"] for r in recent]
  avg    = statistics.mean(temps)
  trend  = temps[-1] - temps[0]  # rising if positive
  hour   = datetime.now().hour

  # Time-based comfort adjustments
  if 22 <= hour <= 6:   base = 24   # night mode — relaxed
  elif 7 <= hour <= 9:  base = 22   # morning — cool
  elif 10 <= hour <= 17: base = 22  # work hours — maintain
  else:                  base = 23   # evening — slightly warm

  # Temperature-trend anticipation
  if trend > 0.5:   base -= 1   # rising fast → pre-cool
  elif trend < -0.5: base += 1   # falling → relax setpoint

  fan = 40 if avg < base + 1 else 80  # ramp fan with load
  return {"setpoint": base, "fan": fan, "source": "cloud_opt"}

# ── MQTT callbacks ─────────────────────────────────────
def on_connect(client, userdata, flags, rc):
  print(f"MQTT connected (rc={rc})")
  client.subscribe(TOPIC_SUB)

def on_message(client, userdata, msg):
  global publish_cnt
  try:
    data = json.loads(msg.payload)
    data["ts"] = datetime.now().isoformat()
    readings.append(data)
    publish_cnt += 1

    print(f"[SENSOR] T={data['temp']}°C H={data['hum']}%")

    # Optimize after every N readings
    if publish_cnt % PUBLISH_EVERY == 0 and len(readings) >= WINDOW_SIZE:
      cmd = optimize_setpoint(list(readings)[-WINDOW_SIZE:])
      client.publish(TOPIC_CMD, json.dumps(cmd))
      print(f"[CLOUD→ESP32] Optimized: {cmd}")

  except Exception as e:
    print(f"Message error: {e}")

# ── Flask REST API ─────────────────────────────────────
app = Flask(__name__)
CORS(app)

@app.route("/api/readings")
def get_readings():
  n = int(request.args.get("n", 50))
  return jsonify(list(readings)[-n:])

@app.route("/api/command", methods=["POST"])
def send_command():
  cmd = request.json
  mqtt_client.publish(TOPIC_CMD, json.dumps(cmd))
  return jsonify({"status": "sent", "cmd": cmd})

@app.route("/api/latest")
def latest():
  return jsonify(readings[-1] if readings else {})

# ── Main entry ─────────────────────────────────────────
def start_mqtt():
  global mqtt_client
  mqtt_client = mqtt.Client()
  mqtt_client.on_connect = on_connect
  mqtt_client.on_message = on_message
  mqtt_client.connect(BROKER_HOST, BROKER_PORT)
  mqtt_client.loop_forever()

if __name__ == "__main__":
  t = threading.Thread(target=start_mqtt, daemon=True)
  t.start()
  print("Cloud server starting on http://0.0.0.0:5000")
  app.run(host="0.0.0.0", port=5000)
